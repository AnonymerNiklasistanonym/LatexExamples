% Styled code input
% \codefile[linenos, highlightlines={10-12}, firstline=5]{python}{code/client.py}
\NewDocumentCommand{\codefile}{O{} m m}{%
	\inputminted[
		fontsize=\normalsize,
		baselinestretch=1.2,
		breaklines,
		frame=lines,
		rulecolor=\color{codeframe},
		framesep=2mm,
		xleftmargin=5mm,
		xrightmargin=5mm,
		#1
	]{#2}{#3}%
}
